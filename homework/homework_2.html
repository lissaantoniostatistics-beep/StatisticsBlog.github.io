Implementazione Cifrario di Cesare in Python

# Inserire qui il codice per il cifrario di Cesare
def cifra_cesare(testo, shift):
    risultato = ""
    for char in testo:
        if 'a' <= char <= 'z':
            risultato += chr(((ord(char) - ord('a') + shift) % 26) + ord('a'))
        elif 'A' <= char <= 'Z':
            risultato += chr(((ord(char) - ord('A') + shift) % 26) + ord('A'))
        else:
            risultato += char
    return risultato

testo_originale = "Ciao mondo"
testo_cifrato = cifra_cesare(testo_originale, 3)
print(f"Testo cifrato: {testo_cifrato}")

Analisi di frequenza delle lettere

# Inserire qui il codice per l'analisi di frequenza
from collections import Counter

def analizza_frequenza(testo):
    frequenze = Counter(c for c in testo.lower() if 'a' <= c <= 'z')
    totale_lettere = sum(frequenze.values())
    distribuzione = {char: count / totale_lettere for char, count in frequenze.items()}
    return distribuzione

testo_cifrato_esempio = "fldb prqgr"
dist_cifrato = analizza_frequenza(testo_cifrato_esempio)
print(f"Distribuzione lettere testo cifrato: {dist_cifrato}")
