üíª Frammenti di Codice

Codice Python utilizzato per l'analisi statistica descrittiva:

Caricamento e prime esplorazioni

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from scipy import stats

# Caricamento del dataset
df = pd.read_csv('student_performance.csv')

# Informazioni generali
print("Forma del dataset:", df.shape)
print("\nPrime 5 righe:")
print(df.head())

# Verifica valori mancanti
print("\nValori mancanti per colonna:")
print(df.isnull().sum())

Statistiche descrittive

# Statistiche descrittive per variabili numeriche
numeric_cols = ['math_score', 'reading_score', 'writing_score']

# Statistiche base
print("Statistiche descrittive:")
print(df[numeric_cols].describe())

# Calcolo manuale di alcune statistiche
for col in numeric_cols:
    print(f"\n--- {col.upper()} ---")
    print(f"Media: {df[col].mean():.2f}")
    print(f"Mediana: {df[col].median():.2f}")
    print(f"Moda: {df[col].mode().values[0]:.2f}")
    print(f"Deviazione Standard: {df[col].std():.2f}")
    print(f"Varianza: {df[col].var():.2f}")
    print(f"Range: {df[col].max() - df[col].min():.2f}")

Identificazione outlier

# Metodo IQR per identificare outlier
def find_outliers_iqr(data):
    Q1 = data.quantile(0.25)
    Q3 = data.quantile(0.75)
    IQR = Q3 - Q1
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    return data[(data < lower_bound) | (data > upper_bound)]

# Identificazione outlier per matematica
outliers_math = find_outliers_iqr(df['math_score'])
print(f"Outlier in matematica: {len(outliers_math)} studenti")
print(f"Valori: {sorted(outliers_math.values)}")

# Visualizzazione outlier
plt.figure(figsize=(10, 6))
plt.boxplot(df['math_score'])
plt.title('Box Plot - Punteggi Matematica con Outlier')
plt.ylabel('Punteggio')
plt.show()
